        \begin{align*}
          u(x,0)=
          \begin{cases}
              0 &\text{if }x<0\\
              1 &\text{if }x>0
          \end{cases}&&u_{t}\left(\frac{u^{2}}{2}\right)_{x}=0
        \end{align*}
        Thus $f'(u)=u$ $\implies (f')^{-1}(u)=u$ s.t.:
        \begin{align*}
          u(x,t)=
          \begin{cases}
              0 &\text{if }x<0\\
              \frac{x}{t} &\text{if }x>0\\
              1 &\text{if }x>0
          \end{cases}
        \end{align*}
