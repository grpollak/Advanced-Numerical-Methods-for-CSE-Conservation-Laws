\begin{defnbox}\nospacing
    \begin{defn}[Conservative Schemes]\label{defn:conservative_schemes}
        \begin{align}
          \sum_{\idxj}U_{\idxj}^{\idxn+1}=\sum_{\idxj}U_{\idxj}^{\idxn}
        \end{align}
    \end{defn}
\end{defnbox}
\begin{corbox}\nospacing
    \begin{cor}[FVS are conservative]\label{cor:fvs_are_conservative}
        FVM schemes\cref{defn:finite_volume_scheme} are conservative.
    \end{cor}
\end{corbox}
