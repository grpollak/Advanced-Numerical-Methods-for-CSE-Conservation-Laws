\begin{defnbox}\nospacing
    \begin{defn}[\hfill\proofref{proof:exact_flux_for_conservation_laws}\newline Finite Volume Scheme for Linear Systems]\label{defn:newline_finite_volume_scheme_for_linear_systems}\leavevmode\\
\begin{circlelistnosep}
    \item \tc{section}{\imp{Reconstruction}}:
    \begin{align*}
      \Uvec(x,t^{n})=\polynomial_{\idxj}^{n}(x)
      \eqs{i.e.}
      \begin{cases}
        \Uvec_{\idxj}^{n}&\text{p.w.\ const}\\
        \Uvec_{\idxj}^{n}=\Uvec_{\idxj}^{n}\pm\frac{\Delta x}{2}\sigmac_{\idxj}^{n}&\text{linear}\\
    \end{cases}\\[-1\jot]
            x_{\idxj-1/2}\leq x \leq x_{\idxj+1/2}
    \end{align*}
    \item \tc{section}{\imp{Evolution}}: by solving Riemann problems:
    \begin{align*}
          &\Uvec_{t}+\linfluxvec\Uvec_{x}=0 \\[-1\jot]
          &\Uvec\left(x,t^{n}\right)=
          \begin{cases}
              \Uvec_{\idxj}^{n}&\text{if }x<x_{\idxj+1/2}\\
              \Uvec_{\idxj+1}^{n}&\text{if }x>x_{\idxj+1/2}\\
          \end{cases}
    \end{align*}
    \item \tc{section}{\imp{Averaging}}:
    % \begin{align*}
    %   \Uvec_{\idxj}^{n+1}=\frac{1}{\Delta x}\int_{t^{n}}^{t^{n+1}}\Uvec(x,t^{n+1})\diff x
    % \end{align*}
    \begin{align*}
      \Uvec_{\idxj}^{n+1}&=\Uvec_{\idxj}^{n}-\frac{\Delta t}{\Delta x}\left(\Fluxvec^{n}_{\idxj+1/2}-\Fluxvec^{n}_{\idxj-1/2}\right)
    \end{align*}
\end{circlelistnosep}
    \begin{align*}
        \Fluxvec^{n}_{\idxj\pm1/2}=\Fluxvec\left(\Uvec_{\idxj}^{n},\Uvec_{\idxj+1}^{n}\right)=\linfluxvec_{\idxj\pm1/2}\Uvec(x_{\idxj\pm1/2},t^{n})
    \end{align*}
    \end{defn}
\end{defnbox}
